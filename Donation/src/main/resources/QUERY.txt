CREATE SCHEMA CARE;

CREATE TABLE CARE.USERPROFILE(ID NUMBER(4)  AUTO_INCREMENT PRIMARY KEY,
						FIRSTNAME VARCHAR2(50) NOT NULL,
						LASTNAME VARCHAR2(50) NOT NULL,
						PHONENO VARCHAR2(20) NOT NULL,
						EMAILID VARCHAR2(50) UNIQUE NOT NULL,
						PWD VARCHAR2(50) NOT NULL
						);
CREATE TABLE CARE.ADDRESS(ID NUMBER(4) AUTO_INCREMENT PRIMARY KEY,
					LINE VARCHAR2(100),
					CITY VARCHAR2(50),
					STATE VARCHAR2(50),
					ZIPCODE VARCHAR2(20),
					PID NUMBER(4)
					);
					
CREATE TABLE CARE.DONATED_MEDICINE(ID NUMBER(4) AUTO_INCREMENT PRIMARY KEY,
					NAME VARCHAR2(50),
					TYPE VARCHAR2(50),
					POWER VARCHAR2(10),
					STATUS VARCHAR2(20),
					QUANTITY NUMBER(4),
					EXPIRYDATE DATE,
					RECIEVEDDATE DATE,
					DONATEDDATE DATE,
					DONATEDBY NUMBER(4)
					);
CREATE TABLE CARE.REQUESTED_MEDICINE(ID NUMBER(4) AUTO_INCREMENT PRIMARY KEY,
					NAME VARCHAR2(50),
					TYPE VARCHAR2(50),
					POWER VARCHAR2(10),
					STATUS VARCHAR2(20),
					QUANTITY NUMBER(4),
					DONATEDDATE DATE,
					REQUESTEDDATE DATE,
					REQUESTEDBY NUMBER(4)
					);
					
CREATE TABLE CARE.ADMINPROFILE(ID NUMBER(4)  AUTO_INCREMENT PRIMARY KEY,
						FIRSTNAME VARCHAR2(50) NOT NULL,
						LASTNAME VARCHAR2(50) NOT NULL,
						PHONENO VARCHAR2(20) NOT NULL,
						EMAILID VARCHAR2(50) NOT NULL,
						UN VARCHAR2(50) NOT NULL,
						PD VARCHAR2(50) NOT NULL
						);
